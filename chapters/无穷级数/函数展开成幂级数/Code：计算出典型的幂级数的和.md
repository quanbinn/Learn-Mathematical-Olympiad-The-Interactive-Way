```python
import math

n = 11
sum = 0

for i in range(1,n):
    sum = 1 + (1/2)*i - (1/8)*i ** 2 + (1/16)*i**3 - (5/128)*i**4 + (7/256)*i**5 - (27/1024)*i**6
    print(i)
    print(math.sqrt(1+i))
    print(sum)
```
